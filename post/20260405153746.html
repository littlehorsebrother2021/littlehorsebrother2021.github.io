前言

 2026.4.5，经过一个晚上的折腾，把STM32N6的RGB_LCD触摸例程搞定了，现在开始记录一下开发过程。这部分较为简单，快速记录。

正文

1. 硬件准备

 STM32N6开发板，ST-LINK，电源线，数据线(用于串口输出提示)，正点原子通用RGB屏幕(博主的具体型号:LCD主控:0x4384 型号代码ATK-MD0430R-800480，触摸主控型号:GT911，I2C通信)。

2. 软件准备

 STM32CubeIDE，STM32CubeMX，RGB屏幕驱动库(正点原子官方)，串口驱动库，RGB屏幕触摸库(正点原子官方)。

3. 2 STM32CubeIDE

基础操作

  根据我们上一期讲的内容，我将基础操作转换成以下几个内容，直接对照着看你做完了没有即可。

​ 必要操作：

      1. 修改两个链接文件(尾部为RAMxpi1和ROMxpi2_RAMxpi1的两个文件)，修改内容为Memorys 和 Sectors 字段
      2. 将两个链接放进编译选项中，放入后Debug模式下尾部为RAMxpi1的文件高亮。
      3. 引入BSP文件夹📂：必须包含两个必要文件夹中的驱动——SYS和HyperRAM文件夹。保证Debug模式下时钟和内存系统正常运转。
      4. 引入BSP路径：保证你的BSP头文件能被编译器正确找到。
      5. main函数中的框架编写：四个必要的`ifdef DEBUG`定义字段，第一个用于引入SYS和HyperRAM头文件，第二个用于设定HyperRAM变量，第三个用于设定时钟，最后一个用于将外部RAM并入整体内存框架。
      6. 注释掉cubemx自己生成的MX_XSPI_Init()字段，因为第五步已经做过了。
      7. 在debug模式下Build🔨，🔘设定好RUN运行参数，将烧录规则External loaders填充好。
      8. 烧录，如果没有问题，你就得到了一个最最基础没有配置任何其他外设，可烧录运行的工程。

添加触摸驱动

​ 这部分根据你的触摸IC进行分门别类，博主的是GT911电容式触摸屏，需要的正点原子例程代码经过删改后如下：

//rgb_lcd.c
/**
 * @brief   清空屏幕并在右上角显示"RST"
 * @param   无
 * @retval  无
 */
void load_draw_RST(void)
{
  rgblcd_clear(WHITE);       /* 清屏 */
  rgblcd_show_string(rgblcddev.width - 24, 0, 200, 16, 16, "RST", BLUE); /* 显示清屏区域 */
}

/**
 * @brief   画粗线
 * @param   x1,y1: 起点坐标
 * @param   x2,y2: 终点坐标
 * @param   size : 线条粗细程度
 * @param   color: 线的颜色
 * @retval  无
 */
void lcd_draw_bline(uint16_t x1, uint16_t y1, uint16_t x2, uint16_t y2, uint8_t size, uint16_t color)
{
  uint16_t t;
  int xerr = 0, yerr = 0, delta_x, delta_y, distance;
  int incx, incy, row, col;

  if (x1 < size || x2 < size || y1 < size || y2 < size)return;

  delta_x = x2 - x1;  /* 计算坐标增量 */
  delta_y = y2 - y1;
  row = x1;
  col = y1;

  if (delta_x > 0)incx = 1;       /* 设置单步方向 */
  else if (delta_x == 0)incx = 0; /* 垂直线 */
  else
  {
    incx = -1;
    delta_x = -delta_x;
  }

  if (delta_y > 0)
  {
    incy = 1;
  }
  else if (delta_y == 0)
  {
    incy = 0;   /* 水平线 */
  }
  else
  {
    incy = -1;
    delta_y = -delta_y;
  }

  if ( delta_x > delta_y)distance = delta_x;  /* 选取基本增量坐标轴 */
  else distance = delta_y;

  for (t = 0; t <= distance + 1; t++ )        /* 画线输出 */
  {
    rgblcd_fill_circle(row, col, size, color); /* 画点 */
    xerr += delta_x ;
    yerr += delta_y ;

    if (xerr > distance)
    {
      xerr -= distance;
      row += incx;
    }

    if (yerr > distance)
    {
      yerr -= distance;
      col += incy;
    }
  }
}

//touch.c
/**
 * @brief       触摸屏初始化
 * @param       无
 * @retval      0,没有进行校准
 *              1,进行过校准
 */
uint8_t tp_init(void)
{
    GPIO_InitTypeDef gpio_init_struct;
    
    tp_dev.touchtype = 0;                   /* 默认设置(电阻屏 & 竖屏) */
    tp_dev.touchtype |= rgblcddev.dir & 0X01;  /* 根据LCD判定是横屏还是竖屏 */

    if (rgblcddev.id == 0x7796)                /* 3.5寸屏有两种，一种屏幕ID为0x5510带电阻触摸屏，一种屏幕ID为0x7796带GT型号的电容触摸屏 */
    {
        if (gt9xxx_init() == 0)             /* 初始化GT系列触摸屏成功,即当前3.5寸屏为电容触摸屏 */
        {
            tp_dev.scan = gt9xxx_scan;      /* 扫描函数指向GT9147触摸屏扫描 */
            tp_dev.touchtype |= 0X80;       /* 电容屏 */
            return 0;
        }
    }

    if (rgblcddev.id == 0X5510 || rgblcddev.id == 0X9806 || rgblcddev.id == 0X4342 || rgblcddev.id == 0X4384 || rgblcddev.id == 0X1018)  /* 电容触摸屏,4.3寸/10.1寸屏 */
    {
        gt9xxx_init();
        tp_dev.scan = gt9xxx_scan;      /* 扫描函数指向GT9147触摸屏扫描 */
        tp_dev.touchtype |= 0X80;       /* 电容屏 */
        return 0;
    }
    else if (rgblcddev.id == 0X1963 || rgblcddev.id == 0X7084 || rgblcddev.id == 0X7016)     /* SSD1963 7寸屏或者 7寸800*480/1024*600 RGB屏 */
    {
        if (!ft5206_init())             /* 触摸IC是FT系列的就执行ft5206_init函数以及使用ft5206_scan扫描函数 */
        {
            tp_dev.scan = ft5206_scan;  /* 扫描函数指向FT5206触摸屏扫描 */
        }
        else                            /* 触摸IC是GT系列的就执行gt9xxx_init函数以及使用gt9xxx_scan扫描函数 */
        {
            gt9xxx_init();
            tp_dev.scan = gt9xxx_scan;  /* 扫描函数指向GT9147触摸屏扫描 */
        }
        tp_dev.touchtype |= 0X80;   /* 电容屏 */
        return 0;
    }
    else
    {
        T_PEN_GPIO_CLK_ENABLE();    /* T_PEN脚时钟使能 */
        T_CS_GPIO_CLK_ENABLE();     /* T_CS脚时钟使能 */
        T_MISO_GPIO_CLK_ENABLE();   /* T_MISO脚时钟使能 */
        T_MOSI_GPIO_CLK_ENABLE();   /* T_MOSI脚时钟使能 */
        T_CLK_GPIO_CLK_ENABLE();    /* T_CLK脚时钟使能 */

        gpio_init_struct.Pin = T_PEN_GPIO_PIN;
        gpio_init_struct.Mode = GPIO_MODE_INPUT;                 /* 输入 */
        gpio_init_struct.Pull = GPIO_PULLUP;                     /* 上拉 */
        gpio_init_struct.Speed = GPIO_SPEED_FREQ_VERY_HIGH;      /* 高速 */
        HAL_GPIO_Init(T_PEN_GPIO_PORT, &gpio_init_struct);       /* 初始化T_PEN引脚 */

        gpio_init_struct.Pin = T_MISO_GPIO_PIN;
        HAL_GPIO_Init(T_MISO_GPIO_PORT, &gpio_init_struct);      /* 初始化T_MISO引脚 */

        gpio_init_struct.Pin = T_MOSI_GPIO_PIN;
        gpio_init_struct.Mode = GPIO_MODE_OUTPUT_PP;             /* 推挽输出 */
        gpio_init_struct.Pull = GPIO_PULLUP;                     /* 上拉 */
        gpio_init_struct.Speed = GPIO_SPEED_FREQ_VERY_HIGH;      /* 高速 */
        HAL_GPIO_Init(T_MOSI_GPIO_PORT, &gpio_init_struct);      /* 初始化T_MOSI引脚 */

        gpio_init_struct.Pin = T_CLK_GPIO_PIN;
        HAL_GPIO_Init(T_CLK_GPIO_PORT, &gpio_init_struct);       /* 初始化T_CLK引脚 */

        gpio_init_struct.Pin = T_CS_GPIO_PIN;
        HAL_GPIO_Init(T_CS_GPIO_PORT, &gpio_init_struct);        /* 初始化T_CS引脚 */

        tp_read_xy(&tp_dev.x[0], &tp_dev.y[0]); /* 第一次读取初始化 */
        eeprom_init();                         /* 初始化24CXX */

        if (tp_get_adjust_data())
        {
            return 0;                           /* 已经校准 */
        }
        else                                    /* 未校准? */
        {
            rgblcd_clear(WHITE);                /* 清屏 */
            tp_adjust();                        /* 屏幕校准 */
            tp_save_adjust_data();
        }

        tp_get_adjust_data();
    }

    return 1;
}








/* 10个触控点的颜色(电容触摸屏用) */
const uint16_t POINT_COLOR_TBL[10] = {RED, GREEN, BLUE, BROWN, YELLOW, MAGENTA, CYAN, LIGHTBLUE, BRRED, GRAY};

/**
 * @brief       电容触摸屏测试函数
 * @param       无
 * @retval      无
 */
void ctp_test(void)
{
  uint8_t t = 0;
  uint8_t i = 0;
  uint16_t lastpos[10][2];        /* 最后一次的数据 */
  uint8_t maxp = 5;

  if (rgblcddev.id == 0X1018)maxp = 10;

  while (1)
  {
    tp_dev.scan(0);

    for (t = 0; t < maxp; t++)
    {
      if ((tp_dev.sta) & (1 << t))
      {
        if (tp_dev.x[t] < rgblcddev.width && tp_dev.y[t] < rgblcddev.height)  /* 坐标在屏幕范围内 */
        {
          if (lastpos[t][0] == 0XFFFF)
          {
            lastpos[t][0] = tp_dev.x[t];
            lastpos[t][1] = tp_dev.y[t];
          }

          lcd_draw_bline(lastpos[t][0], lastpos[t][1], tp_dev.x[t], tp_dev.y[t], 2, POINT_COLOR_TBL[t]); /* 画线 */
          lastpos[t][0] = tp_dev.x[t];
          lastpos[t][1] = tp_dev.y[t];

          if (tp_dev.x[t] > (rgblcddev.width - 24) && tp_dev.y[t] < 20)
          {
            load_draw_RST();/* 清除 */
          }
        }
      }
      else
      {
        lastpos[t][0] = 0XFFFF;
      }
    }

    HAL_Delay(10);
    i++;

  }
}
